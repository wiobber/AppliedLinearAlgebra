%matrix_symmetry
\begin{problem}
  Consider the matrices
  \begin{equation*}
    A =\begin{mymatrix}{rr}
      1 & 2 \\
      3 & 2 \\
      1 & -1
    \end{mymatrix},
    \quad
    B=\begin{mymatrix}{rrr}
      2 & -5 & 2 \\
      -3 & 2 & 1
    \end{mymatrix}
  \end{equation*}
  Find the following if possible. If it is not possible explain why.
  \begin{enumerate}
  \item $-3A{^T}$.
  \item $3B - A^T$.
  \end{enumerate}

  \begin{sol}
    \begin{enumerate}
    \item $\begin{mymatrix}{rrr}
        -3 & -9 & -3 \\
        -6 & -6 & 3
      \end{mymatrix}$.
    \item $\begin{mymatrix}{rrr}
        5 & -18 & 5 \\
        -11 & 4 & 4
      \end{mymatrix}$.
    \end{enumerate}
  \end{sol}

  Which of the following matrices are symmetric, antisymmetric, both,
  or neither?
  \begin{equation*}
    A = \begin{mymatrix}{rr}
      0 & 1 \\
      -1 & 0 \\
    \end{mymatrix},
    \quad
    B = \begin{mymatrix}{rr}
      2 & 1 \\
      1 & 3 \\
    \end{mymatrix},
    \quad
    C = \begin{mymatrix}{rr}
      1 & 2 \\
      -2 & 0 \\
    \end{mymatrix},
    \quad
    D = \begin{mymatrix}{rr}
      0 & 0 \\
      0 & 0 \\
    \end{mymatrix}.
  \end{equation*}
  \begin{sol}
    $A$ is antisymmetric, $B$ is symmetric, $C$ is neither, and $D$ is both.
  \end{sol}
\end{problem}
