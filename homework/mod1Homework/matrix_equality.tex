%matrix_equality
\begin{problem}
  Find scalars $x,y,z$ such that the following two matrices are equal.
  \begin{equation*}
    \begin{mymatrix}{rr}
      x & -1 \\
      2 & 4
    \end{mymatrix}
    \quad\mbox{and}\quad
    \begin{mymatrix}{rr}
      2 & y \\
      z & 4
    \end{mymatrix}.
  \end{equation*}
  \begin{sol}
    $x=\answer{2}$, $y=\answer{-1}$, $z=\answer{2}$.
  \end{sol}
\end{problem}
