%matrix_equality
\begin{problem}
  Find scalars $x,y,z$ such that the following two matrices are equal.

  \begin{hint}

    Two matrices are equal if and only if their corresponding entries are equal. 

  \end{hint}
  \begin{equation*}
    \startmat{cc}
      x & -1 \\
      2 & 4
    \stopmat
    \quad\mbox{and}\quad
    \startmat{cc}
      2 & y \\
      z & 4
    \stopmat.
  \end{equation*}

    $x=\answer{2}$, $y=\answer{-1}$, $z=\answer{2}$.

\end{problem}
