\begin{exercise}

  Which of the following equations are linear?
  \begin{selectAll}
  \choice[correct]{$2x-3y+4z = -10$}
  \choice[correct]{$2.123x_1 + 5.541x_2 - 9.101x_3 = 11.012$}
  \choice{$x^2+y^2+z^2 = 1$}
  \choice[correct]{$\frac{1}{\sqrt{2}}x+4^3y = \sin(\frac{\pi}{3})$}
  \choice{$x+yz = 3$}
  \end{selectAll}

  \begin{hint}
    Remember that equations are linear if they are linear combinations of variables (raised to the first power) and constants.
  \end{hint}

\end{exercise}
