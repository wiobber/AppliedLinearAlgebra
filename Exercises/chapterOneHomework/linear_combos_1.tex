\begin{exercise}

\begin{equation*}
  \vect{v}= \startmat{r}
    4 \\
    4 \\
    -3
  \stopmat
\end{equation*}
is a linear combination of the vectors
\begin{equation*}
  \vect{u}_1 = \startmat{r}
    3 \\
    1 \\
    -1
  \stopmat
  \quad\mbox{and}\quad
  \vect{u}_2 =
  \startmat{r}
    2 \\
    -2 \\
    1
  \stopmat.
\end{equation*}

  Find scalars $a$ and $b$ so that $\vect{v} = a\vect{u}_1 + b\vect{u}_2$.
  \begin{align*}
    a &= \answer{2} \\
    b &= \answer{-1}
  \end{align*}

\end{exercise}
