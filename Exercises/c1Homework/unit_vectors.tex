\begin{exercise}

Which of the following are unit vectors?
\begin{equation*}
  \vect{u}=\frac{1}{2}\startmat{c}1\\1\\1\stopmat, \quad
  \vect{v}=\frac{1}{3}\startmat{c}1\\1\\1\stopmat, \quad
  \vect{w}=\frac{1}{2}\startmat{c}1\\1\\1\\1\stopmat.
\end{equation*}

$\vect{u}$ \wordChoice{
  \choice{is}
  \choice[correct]{isn't}
  } a unit vector with norm $\answer{\frac{\sqrt{3}}{2}}$. $\vect{v}$ \wordChoice{
    \choice{is}
    \choice[correct]{isn't}
    } a unit vector with norm $\answer{\frac{\sqrt{3}}{3}}$. $\vect{w}$ \wordChoice{
      \choice[correct]{is}
      \choice{isn't}
      } a unit vector with norm $\answer{1}$.

\end{exercise}
