\begin{problem}

    Consider the system of equations
    \begin{equation*}
      \begin{array}{c@{~}c@{~}c@{~}c@{~}c@{~}c@{~}c@{~}c@{~}c}
        x&+&2y&+&3z&+&4w &=& 4 \\
        x&+&y&+&z&+&w &=& 2 \\
        x&+&2y&+&2z&+&w &=& 2.
      \end{array}
    \end{equation*}
    For each of the following tuples $(x,y,z,w)$ of real numbers,
    determine whether it is a solution of the first equation, second
    equation, and/or third equation. Which ones are solutions to the
    system of equations?
  
    \begin{hint}
  
      It might be faster to translate the system into a matrix and then treat each tuple as a vector, checking whether the matrix times the vector equals the right hand side of the system of equations.
  
    \end{hint}
  
    \begin{enumerate}
        \item $(2,0,-2,2)$
        \begin{selectAll}
          \choice[correct]{Satisfies Equation 1}
          \choice[correct]{Satisfies Equation 2}
          \choice{Satisfies Equation 3}
        \end{selectAll}
      
        \item $(2,2,-2,0)$
        \begin{selectAll}
          \choice{Satisfies Equation 1}
          \choice[correct]{Satisfies Equation 2}
          \choice[correct]{Satisfies Equation 3}
        \end{selectAll}
      
        \item $(1,1,-1,1)$
        \begin{selectAll}
          \choice[correct]{Satisfies Equation 1}
          \choice[correct]{Satisfies Equation 2}
          \choice[correct]{Satisfies Equation 3}
        \end{selectAll}
      
        \item $(3,0,-1,1)$
        \begin{selectAll}
          \choice[correct]{Satisfies Equation 1}
          \choice{Satisfies Equation 2}
          \choice[correct]{Satisfies Equation 3}
        \end{selectAll}
      
        \item $(2,-2,2,0)$
        \begin{selectAll}
          \choice[correct]{Satisfies Equation 1}
          \choice[correct]{Satisfies Equation 2}
          \choice[correct]{Satisfies Equation 3}
        \end{selectAll}
      \end{enumerate}    
      
      Which of the following tuples are solutions to the system of equations?

        \begin{selectAll}
            \choice{$(2,0,-2,2)$}
            \choice{$(2,2,-2,0)$}
            \choice[correct]{$(1,1,-1,1)$}
            \choice{$(3,0,-1,1)$}
            \choice[correct]{$(2,-2,2,0)$}
        \end{selectAll}

\end{problem}
